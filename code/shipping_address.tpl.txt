<script type="text/javascript"><!--
$('#shipping-address input[name=\'shipping_address\']').live('change', function() {
	if (this.value == 'new') {
		$('#shipping-existing').hide();
		$('#shipping-new').show();
	} else if (this.value == 'existing') {
		$('#shipping-existing').show();
		$('#shipping-new').hide();
	} else {
		$('#shipping-existing').hide();
		$('#shipping-new').hide();
	}
});
//--></script>
